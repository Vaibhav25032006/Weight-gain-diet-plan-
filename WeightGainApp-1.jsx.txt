// ─── STORAGE HELPERS ───────────────────────────────────────────────────────────
const STORAGE_KEY = "wgdp_data_v2";
function loadData() {
  try {
    const raw = localStorage.getItem(STORAGE_KEY);
    return raw ? JSON.parse(raw) : { completions: {}, lang: "en", streak: 0, lastDate: null };
  } catch { return { completions: {}, lang: "en", streak: 0, lastDate: null }; }
}
function saveData(d) {
  try { localStorage.setItem(STORAGE_KEY, JSON.stringify(d)); } catch {}
}

// ─── DATE HELPERS ──────────────────────────────────────────────────────────────
function getDateKey(date) {
  return `${date.getFullYear()}-${String(date.getMonth()+1).padStart(2,'0')}-${String(date.getDate()).padStart(2,'0')}`;
}

// ─── MAIN APP ──────────────────────────────────────────────────────────────────
export default function App() {
  const [store, setStore] = useState(loadData);
  const [tab, setTab] = useState("today");
  const [now, setNow] = useState(new Date());
  const [aiMessages, setAiMessages] = useState([]);
  const [aiInput, setAiInput] = useState("");
  const [aiLoading, setAiLoading] = useState(false);
  const [aiMode, setAiMode] = useState("chat"); // chat | meal | analyze | motivate
  const [shareToast, setShareToast] = useState(false);
  const [confetti, setConfetti] = useState([]);
  const [celebTask, setCelebTask] = useState(null);

  const lang = store.lang || "en";
  const t = T[lang];
  const tasks = TASKS_DATA[lang];
  const todayKey = getDateKey(now);

  // Clock tick
  useEffect(() => {
    const interval = setInterval(() => setNow(new Date()), 60000);
    return () => clearInterval(interval);
  }, []);

  // Persist
  useEffect(() => { saveData(store); }, [store]);

  // Streak calc
  useEffect(() => {
    const yesterday = new Date(now);
    yesterday.setDate(yesterday.getDate() - 1);
    const yk = getDateKey(yesterday);
    if (store.lastDate && store.lastDate !== todayKey) {
      const completions = store.completions[yk] || {};
      const rate = Object.values(completions).filter(Boolean).length / tasks.length;
      if (rate >= 0.7) {
        if (store.lastDate === yk) {
          setStore(s => ({...s, streak: (s.streak||0)+1, lastDate: todayKey}));
        }
      } else if (store.lastDate !== yk) {
        setStore(s => ({...s, streak: 0, lastDate: todayKey}));
      }
    }
  }, [todayKey]);

  const todayCompletions = store.completions[todayKey] || {};
  const completedCount = Object.values(todayCompletions).filter(Boolean).length;
  const totalTasks = tasks.length;
  const pct = Math.round((completedCount / totalTasks) * 100);

  function toggleTask(id) {
    const newVal = !todayCompletions[id];
    const newCompletions = { ...store.completions, [todayKey]: { ...todayCompletions, [id]: newVal } };
    setStore(s => ({...s, completions: newCompletions, lastDate: todayKey}));
    if (newVal) {
      setCelebTask(id);
      launchConfetti();
      setTimeout(() => setCelebTask(null), 1000);
    }
  }

  function launchConfetti() {
    const pieces = Array.from({length: 18}, (_, i) => ({
      id: i, x: Math.random()*100, color: ["#FFD700","#FF6B6B","#4FC3F7","#81C784","#CE93D8"][i%5],
      size: Math.random()*8+4, delay: Math.random()*0.5
    }));
    setConfetti(pieces);
    setTimeout(() => setConfetti([]), 2000);
  }

  function resetToday() {
    setStore(s => ({...s, completions: {...s.completions, [todayKey]: {}}}));
  }

  // Weekly data
  function getWeekData() {
    const result = [];
    for (let i = 6; i >= 0; i--) {
      const d = new Date(now);
      d.setDate(d.getDate() - i);
      const key = getDateKey(d);
      const comp = store.completions[key] || {};
      const done = Object.values(comp).filter(Boolean).length;
      result.push({ day: t.weekdays[d.getDay()], pct: Math.round((done/totalTasks)*100), done, date: key });
    }
    return result;
  }

  // Monthly data
  function getMonthData() {
    const result = [];
    const daysInMonth = new Date(now.getFullYear(), now.getMonth()+1, 0).getDate();
    let totalPct = 0, counted = 0;
    for (let i = 1; i <= daysInMonth; i++) {
      const d = new Date(now.getFullYear(), now.getMonth(), i);
      const key = getDateKey(d);
      const comp = store.completions[key] || {};
      const done = Object.values(comp).filter(Boolean).length;
      const p = Math.round((done/totalTasks)*100);
      if (d <= now) { totalPct += p; counted++; }
      result.push({ day: i, pct: p, isPast: d <= now });
    }
    return { days: result, avg: counted ? Math.round(totalPct/counted) : 0 };
  }

  // Greeting
  const hour = now.getHours();
  const greetPart = hour < 12 ? t.morning : hour < 17 ? t.afternoon : hour < 21 ? t.evening : t.night;

  // AI Chat
  async function sendAI(customPrompt) {
    const prompt = customPrompt || aiInput.trim();
    if (!prompt && !customPrompt) return;
    const userMsg = { role: "user", content: prompt };
    const newMsgs = [...aiMessages, userMsg];
    setAiMessages(newMsgs);
    setAiInput("");
    setAiLoading(true);
    const weekData = getWeekData();
    const todayDone = completedCount;
    const systemPrompt = `You are a friendly, knowledgeable health and nutrition AI coach specialized in HEALTHY WEIGHT GAIN. The user is following a structured diet plan with these daily tasks: ${tasks.map(t=>t.title).join(", ")}. Today they've completed ${todayDone}/${totalTasks} tasks (${pct}%). Weekly avg: ${weekData.map(d=>d.pct+'%').join(', ')}. Language preference: ${lang === 'hi' ? 'Hindi' : 'English'}. Respond in ${lang === 'hi' ? 'Hindi' : 'English'}. Be encouraging, specific, and practical. Keep responses concise (under 150 words).`;
    try {
      const response = await fetch("https://api.anthropic.com/v1/messages", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({
          model: "claude-sonnet-4-20250514",
          max_tokens: 1000,
          system: systemPrompt,
          messages: newMsgs,
        }),
      });
      const data = await response.json();
      const reply = data.content?.map(c => c.text || "").join("") || "Sorry, I couldn't respond right now.";
      setAiMessages([...newMsgs, { role: "assistant", content: reply }]);
    } catch {
      setAiMessages([...newMsgs, { role: "assistant", content: lang === "hi" ? "क्षमा करें, अभी जवाब नहीं दे सकते।" : "Sorry, couldn't connect right now." }]);
    }
    setAiLoading(false);
  }

  async function triggerAIFeature(mode) {
    setTab("ai");
    setAiMode(mode);
    let prompt;
    const weekData = getWeekData();
    if (mode === "meal") {
      prompt = lang === "hi"
        ? `मेरी आज की ${completedCount}/${totalTasks} कार्य पूर्ण हुए हैं। मुझे एक कस्टम मील प्लान सुझाएं जो मेरे वजन बढ़ाने के लक्ष्य में मदद करे। विशेष रूप से कल के लिए कुछ नए विकल्प बताएं।`
        : `I've completed ${completedCount}/${totalTasks} tasks today. Suggest a custom meal variation for tomorrow that supports healthy weight gain, keeping the plan but adding variety.`;
    } else if (mode === "analyze") {
      prompt = lang === "hi"
        ? `मेरी साप्ताहिक प्रगति: ${weekData.map(d=>d.day+': '+d.pct+'%').join(', ')}। मेरी प्रगति का विस्तृत विश्लेषण करें और सुधार के सुझाव दें।`
        : `My weekly progress: ${weekData.map(d=>d.day+': '+d.pct+'%').join(', ')}. Analyze my progress patterns and give specific actionable advice to improve consistency.`;
    } else if (mode === "motivate") {
      const streak = store.streak || 0;
      prompt = lang === "hi"
        ? `मेरी ${streak} दिन की स्ट्रीक है और आज ${pct}% कार्य पूर्ण किए। मुझे वजन बढ़ाने के सफर में आगे बढ़ने के लिए प्रेरित करें।`
        : `I have a ${streak}-day streak and completed ${pct}% today. Give me powerful, personalized motivation to keep going on my weight gain journey.`;
    }
    if (prompt) sendAI(prompt);
  }

  // Share
  function shareProgress() {
    const weekData = getWeekData();
    const text = lang === "hi"
      ? `🏋️ मेरी वजन बढ़ाने की प्रगति\n\nआज: ${pct}% (${completedCount}/${totalTasks} कार्य)\n🔥 स्ट्रीक: ${store.streak||0} दिन\n\nसाप्ताहिक रिपोर्ट:\n${weekData.map(d=>`${d.day}: ${d.pct}%`).join('\n')}\n\n#वजनबढ़ाओ #हेल्दीलाइफ`
      : `🏋️ My Weight Gain Progress\n\nToday: ${pct}% (${completedCount}/${totalTasks} tasks)\n🔥 Streak: ${store.streak||0} days\n\nWeekly Report:\n${weekData.map(d=>`${d.day}: ${d.pct}%`).join('\n')}\n\n#WeightGain #HealthyLife`;
    
    if (navigator.share) {
      navigator.share({ title: t.shareTitle, text });
    } else {
      navigator.clipboard.writeText(text).then(() => {
        setShareToast(true);
        setTimeout(() => setShareToast(false), 2500);
      });
    }
  }

  const weekData = getWeekData();
  const monthData = getMonthData();

  return (
    <div style={styles.root}>
      {/* Confetti */}
      {confetti.map(c => (
        <div key={c.id} style={{
          position:"fixed", top: "-10px", left: `${c.x}%`,
          width: c.size, height: c.size, borderRadius: "50%",
          background: c.color, zIndex: 9999,
          animation: `confettiFall 1.8s ease-in forwards`,
          animationDelay: `${c.delay}s`, pointerEvents: "none"
        }}/>
      ))}

      {/* Header */}
      <div style={styles.header}>
        <div style={styles.headerBg}/>
        <div style={styles.headerContent}>
          <div style={styles.headerTop}>
            <div>
              <div style={styles.greeting}>{t.greeting} {greetPart}! 🌟</div>
              <div style={styles.appTitle}>{t.appTitle}</div>
            </div>
            <button onClick={() => setStore(s=>({...s, lang: lang==="en"?"hi":"en"}))} style={styles.langBtn}>
              {t.switchLang}
            </button>
          </div>

          {/* Progress Ring */}
          <div style={styles.progressRow}>
            <div style={styles.ringWrap}>
              <svg width="88" height="88" viewBox="0 0 88 88">
                <circle cx="44" cy="44" r="36" fill="none" stroke="rgba(255,255,255,0.2)" strokeWidth="8"/>
                <circle cx="44" cy="44" r="36" fill="none" stroke="#FFD700" strokeWidth="8"
                  strokeDasharray={`${(pct/100)*226.2} 226.2`}
                  strokeLinecap="round" transform="rotate(-90 44 44)"
                  style={{transition:"stroke-dasharray 0.6s ease"}}/>
                <text x="44" y="49" textAnchor="middle" fill="white" fontSize="18" fontWeight="bold">{pct}%</text>
              </svg>
            </div>
            <div style={styles.statsCol}>
              <div style={styles.statItem}>
                <span style={styles.statNum}>{completedCount}/{totalTasks}</span>
                <span style={styles.statLbl}>{t.completedTasks}</span>
              </div>
              <div style={styles.statItem}>
                <span style={styles.statNum}>🔥 {store.streak||0}</span>
                <span style={styles.statLbl}>{t.streakDays}</span>
              </div>
              <div style={styles.statItem}>
                <span style={styles.statNum}>{weekData.length ? Math.round(weekData.reduce((a,d)=>a+d.pct,0)/weekData.length) : 0}%</span>
                <span style={styles.statLbl}>{t.avgCompletion}</span>
              </div>
            </div>
          </div>

          {/* Time */}
          <div style={styles.timeDisplay}>
            {now.toLocaleTimeString(lang==="hi"?"hi-IN":"en-US", {hour:"2-digit", minute:"2-digit"})} · {now.toLocaleDateString(lang==="hi"?"hi-IN":"en-US", {weekday:"long", month:"short", day:"numeric"})}
          </div>
        </div>
      </div>

      {/* Tabs */}
      <div style={styles.tabBar}>
        {[["today","📋",t.today],["weekly","📊",t.weekly],["monthly","📅",t.monthly],["ai","🤖",t.aiCoach]].map(([key,icon,label])=>(
          <button key={key} onClick={()=>setTab(key)} style={{...styles.tabBtn, ...(tab===key?styles.tabActive:{})}}>
            <span style={{fontSize:16}}>{icon}</span>
            <span style={styles.tabLabel}>{label}</span>
          </button>
        ))}
      </div>

      {/* Content */}
      <div style={styles.content}>

        {/* TODAY TAB */}
        {tab === "today" && (
          <div>
            <div style={styles.sectionHeader}>
              <span>{t.todayProgress}</span>
              <button onClick={resetToday} style={styles.resetBtn}>{t.resetDay}</button>
            </div>

            {tasks.map((task, idx) => {
              const done = !!todayCompletions[task.id];
              const isCurrent = celebTask === task.id;
              return (
                <div key={task.id} onClick={()=>toggleTask(task.id)}
                  style={{...styles.taskCard, ...(done?styles.taskDone:{}), ...(isCurrent?styles.taskCelebrate:{})}}
                >
                  <div style={{...styles.taskAccent, background: task.color}}/>
                  <div style={styles.taskTime}>{task.time}</div>
                  <div style={styles.taskEmoji}>{task.emoji}</div>
                  <div style={styles.taskInfo}>
                    <div style={styles.taskTitle}>{task.title}</div>
                    <div style={styles.taskDesc}>{task.desc}</div>
                    <div style={{...styles.taskTag, background: task.color+'33', color: task.color}}>
                      {task.tag}
                    </div>
                  </div>
                  <div style={{...styles.taskCheck, ...(done?styles.taskCheckDone:{})}}>
                    {done ? "✓" : ""}
                  </div>
                </div>
              );
            })}

            {/* Habits */}
            <div style={styles.habitsCard}>
              <div style={styles.habitsTitle}>💧 {t.dailyHabits}</div>
              <p style={styles.habitsText}>{t.hydration}</p>
              <p style={styles.habitsText}>{t.consistency}</p>
            </div>
            <div style={styles.avoidCard}>
              <div style={styles.avoidTitle}>🚫 {t.foodsToAvoid}</div>
              {[t.avoid1,t.avoid2,t.avoid3,t.avoid4].map((a,i)=>(
                <div key={i} style={styles.avoidItem}>⊗ {a}</div>
              ))}
            </div>

            <div style={styles.footer}>
              <span>{t.rightNutrition}</span><span> • </span>
              <span>{t.consistency2}</span><span> • </span>
              <span>{t.betterResults}</span>
            </div>
          </div>
        )}

        {/* WEEKLY TAB */}
        {tab === "weekly" && (
          <div>
            <div style={styles.sectionHeader}><span>{t.weeklyReport}</span></div>
            <div style={styles.reportCard}>
              {weekData.map((d,i) => (
                <div key={i} style={styles.barRow}>
                  <div style={styles.barDay}>{d.day}</div>
                  <div style={styles.barTrack}>
                    <div style={{...styles.barFill, width:`${d.pct}%`, background: d.pct>=80?"#81C784":d.pct>=50?"#FFD700":"#EF9A9A"}}/>
                  </div>
                  <div style={styles.barPct}>{d.pct}%</div>
                </div>
              ))}
            </div>
            <div style={styles.summaryCard}>
              <div style={styles.summaryItem}>
                <span style={styles.summaryNum}>{weekData.filter(d=>d.pct>=80).length}</span>
                <span style={styles.summaryLbl}>🟢 {lang==="hi"?"अच्छे दिन":"Great Days"}</span>
              </div>
              <div style={styles.summaryItem}>
                <span style={styles.summaryNum}>{Math.round(weekData.reduce((a,d)=>a+d.pct,0)/7)}%</span>
                <span style={styles.summaryLbl}>📊 {lang==="hi"?"औसत":"Average"}</span>
              </div>
              <div style={styles.summaryItem}>
                <span style={styles.summaryNum}>{Math.max(...weekData.map(d=>d.pct))}%</span>
                <span style={styles.summaryLbl}>🏆 {lang==="hi"?"सर्वश्रेष्ठ":"Best Day"}</span>
              </div>
            </div>
            <button onClick={shareProgress} style={styles.shareBtn}>
              📤 {t.shareProgress}
            </button>
          </div>
        )}

        {/* MONTHLY TAB */}
        {tab === "monthly" && (
          <div>
            <div style={styles.sectionHeader}>
              <span>{t.monthlyReport}</span>
              <span style={styles.monthAvg}>{t.months[now.getMonth()]} · {monthData.avg}% {lang==="hi"?"औसत":"avg"}</span>
            </div>
            <div style={styles.calGrid}>
              {t.weekdays.map(d => <div key={d} style={styles.calHead}>{d}</div>)}
              {Array.from({length: new Date(now.getFullYear(), now.getMonth(), 1).getDay()}).map((_,i)=>
                <div key={`e${i}`}/>
              )}
              {monthData.days.map(d => (
                <div key={d.day} style={{
                  ...styles.calCell,
                  background: !d.isPast ? "#f5f5f5" : d.pct>=80?"#C8E6C9":d.pct>=50?"#FFF9C4":d.pct>0?"#FFCDD2":"#EEEEEE",
                  border: d.day === now.getDate() ? "2px solid #2E7D32" : "1px solid #eee",
                  color: d.isPast ? "#333" : "#bbb"
                }}>
                  <div style={styles.calDay}>{d.day}</div>
                  {d.isPast && d.pct > 0 && <div style={styles.calPct}>{d.pct}%</div>}
                </div>
              ))}
            </div>
            <div style={styles.calLegend}>
              {[["#C8E6C9","≥80%"],["#FFF9C4","50–79%"],["#FFCDD2","<50%"]].map(([c,l])=>(
                <div key={l} style={styles.legendItem}><div style={{...styles.legendDot,background:c}}/>{l}</div>
              ))}
            </div>
            <button onClick={shareProgress} style={styles.shareBtn}>
              📤 {t.shareProgress}
            </button>
          </div>
        )}

        {/* AI TAB */}
        {tab === "ai" && (
          <div>
            <div style={styles.sectionHeader}><span>🤖 {t.aiCoach}</span></div>

            {/* AI Feature Buttons */}
            <div style={styles.aiFeatures}>
              <button onClick={()=>triggerAIFeature("meal")} style={styles.aiFeatureBtn}>
                <span style={styles.aiFIcon}>🥗</span>
                <span>{t.aiFeature1}</span>
              </button>
              <button onClick={()=>triggerAIFeature("analyze")} style={styles.aiFeatureBtn}>
                <span style={styles.aiFIcon}>📊</span>
                <span>{t.aiFeature2}</span>
              </button>
              <button onClick={()=>triggerAIFeature("motivate")} style={styles.aiFeatureBtn}>
                <span style={styles.aiFIcon}>💪</span>
                <span>{t.aiFeature3}</span>
              </button>
            </div>

            {/* Chat */}
            <div style={styles.chatBox}>
              {aiMessages.length === 0 && (
                <div style={styles.chatEmpty}>
                  <div style={{fontSize:40}}>🤖</div>
                  <div style={{color:"#666", marginTop:8, fontSize:14, textAlign:"center"}}>
                    {lang==="hi" ? "नमस्ते! मैं आपका AI हेल्थ कोच हूं। ऊपर दिए बटन दबाएं या सीधे सवाल पूछें।" : "Hi! I'm your AI health coach. Tap a feature above or ask me anything!"}
                  </div>
                </div>
              )}
              {aiMessages.map((m,i) => (
                <div key={i} style={{...styles.chatMsg, ...(m.role==="user"?styles.chatUser:styles.chatAssistant)}}>
                  {m.role==="assistant" && <div style={styles.chatIcon}>🤖</div>}
                  <div style={{...styles.chatBubble, ...(m.role==="user"?styles.chatBubbleUser:styles.chatBubbleAI)}}>
                    {m.content}
                  </div>
                </div>
              ))}
              {aiLoading && (
                <div style={styles.chatMsg}>
                  <div style={styles.chatIcon}>🤖</div>
                  <div style={styles.chatBubbleAI}><span style={styles.dots}>●●●</span></div>
                </div>
              )}
            </div>

            <div style={styles.chatInputRow}>
              <input
                value={aiInput}
                onChange={e=>setAiInput(e.target.value)}
                onKeyDown={e=>e.key==="Enter"&&sendAI()}
                placeholder={t.askAI}
                style={styles.chatInput}
              />
              <button onClick={()=>sendAI()} style={styles.chatSend} disabled={aiLoading}>
                {t.aiSend}
              </button>
            </div>
          </div>
        )}
      </div>

      {/* Share Toast */}
      {shareToast && (
        <div style={styles.toast}>✅ {t.copied}</div>
      )}

      <style>{`
        @keyframes confettiFall {
          0% { transform: translateY(0) rotate(0deg); opacity: 1; }
          100% { transform: translateY(100vh) rotate(720deg); opacity: 0; }
        }
        @keyframes dots { 0%,80%,100%{opacity:0} 40%{opacity:1} }
        .dots span { animation: dots 1.4s infinite; }
        * { box-sizing: border-box; -webkit-tap-highlight-color: transparent; }
        body { margin: 0; font-family: 'Segoe UI', sans-serif; background: #f0f4f0; }
        ::-webkit-scrollbar { width: 0; }
      `}</style>
    </div>
  );
}
